function euler(n, k=null) {
if (n === 0) return 1;
if (k === null) k = 1+1/n;
return k*euler(n-1,k)}

function fibonacci(n) {
if (n <= 1) return n+1;
k = fibonacci(n-1) + fibonacci(n-2)
return k;}

function binomialis(k, n) {
if (k === 0 || k === n) return 1;
return binomialis(k-1, n-1) + binomialis(k, n-1)}

function binomialis2(f, k=0) {
